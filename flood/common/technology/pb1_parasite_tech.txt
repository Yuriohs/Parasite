			factor = 5

		}

			factor = 2

		}

			factor = 2
			months_passed > 10
		}		
		modifier = {
			factor = 2
			months_passed > 15
		}

			factor = 2
			months_passed > 20
		}

			}
		}
	}
		ai_weight = {
		modifier = {
			factor = 1.25

				area = society
				has_trait = "leader_trait_expertise_biology"
			}
		}
	}
	
	
	## unlocks the ability to build breeder buildings
}
